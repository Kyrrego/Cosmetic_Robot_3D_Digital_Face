#include "3DCamera.hpp"
#include <thread>
int main()
{
printf( "This is SampleSetAndGetUserData sample!\n\n");
ERROR_CODE ret;
// get camera pointer and connect a valid camera
cs::ICameraPtr camera = cs::getCameraPtr();
ret = camera->connect();
if(ret != SUCCESS)
{
printf("camera connect failed(%d)!\n", ret);
return -1;
}
// get informations of camera
CameraInfo info;
ret = camera->getInfo(info);
if(ret != SUCCESS)
{
printf("camera get info failed(%d)!\n", ret);
return -1;
}
// display informations of camera
printf("%20s : %s\n", "name", info.name);
printf("%20s : %s\n", "serial", info.serial);
printf("%20s : %s\n", "unique id", info.uniqueId);
printf("%20s : %s\n", "firmware version", info.firmwareVersion);
printf("%20s : %s\n", "algorithm version", info.algorithmVersion);
printf("\n");
// set user data
char input_data[] = "cameraId:cs01";
ret = camera->setUserData(input_data, strlen(input_data));
if(ret != SUCCESS)
{
printf("camera set user data failed(%d)!\n", ret);
return -1;
}
else
{
printf("set user data success![%s] length=%d\n\n", input_data, (int)(strlen(input_data)));
}
// get user data
char output_data[1024] = {0};
int length = 0;
ret = camera->getUserData(output_data, length);
if(ret != SUCCESS)
{
printf("camera get user data failed(%d)!\n", ret);
return -1;
}
else
{
printf("get user data success [%s] length=%d\n\n", output_data, length);
}
std::this_thread::sleep_for(std::chrono::seconds(5));
// disconnect camera
ret = camera->disconnect();
if(ret != SUCCESS)
{
printf("camera disconnect failed(%d)!\n", ret);
return -1;
}
return 0;
}
